// --- BAGIAN INI MENGGANTIKAN FUNGSI generateImage() LAMA --- async function generateImage() { const prompt = imagePromptInput.value.trim(); const style = imageStyleSelect.value.trim(); const aspectRatio = imageAspectRatioSelect.value.trim(); // imageCount tidak digunakan, karena Gemini hanya menghasilkan 1 gambar. // const imageCount = parseInt(imageCountSelect.value, 10); const referenceImageFile = imageReferenceUpload.files[0]; if (!prompt && !referenceImageFile) { showMessage("Harap masukkan deskripsi atau unggah gambar referensi."); return; } if (apiKey === "") { showMessage("Kesalahan: API Key kosong. Harap masukkan API Key Anda untuk menggunakan fitur ini."); return; } // Show loading state loadingIndicator.classList.remove('hidden'); generateImageBtn.disabled = true; downloadImageBtn.classList.add('hidden'); imageOutputContainer.innerHTML = ''; lastGeneratedImageData = []; try { // SELALU gunakan endpoint Gemini untuk memaksimalkan peluang akses gratis. const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`; let parts = []; let finalPrompt = prompt; if (style) { finalPrompt += `, ${style}`; } if (aspectRatio) { finalPrompt += `, ${aspectRatio}`; } parts.push({ text: finalPrompt }); if (referenceImageFile) { const base64Data = await fileToBase64(referenceImageFile); const imageMimeType = referenceImageFile.type; parts.push({ inlineData: { mimeType: imageMimeType, data: base64Data } }); } const payload = { contents: [{ parts: parts }], generationConfig: { // Konfigurasi penting: Minta model untuk menghasilkan gambar. responseModalities: ["IMAGE"] } }; console.log('Sending request to:', apiUrl); console.log('Payload:', payload); const response = await fetch(apiUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); if (!response.ok) { const errorJson = await response.json(); throw new Error(`HTTP error! status: ${response.status}. Detail: ${errorJson.error?.message || 'Unknown error'}`); } const result = await response.json(); // Ekstraksi data gambar dari respons Gemini const base64Data = result?.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data; if (base64Data) { // Karena Gemini hanya mengembalikan 1 gambar, kita hanya mengambil satu hasil. lastGeneratedImageData = [base64Data]; const imgElement = document.createElement('img'); imgElement.src = `data:image/png;base64,${base64Data}`; imgElement.alt = `Generated Image 1`; imgElement.className = "w-full h-auto rounded-lg shadow-xl border border-gray-600"; imageOutputContainer.appendChild(imgElement); downloadImageBtn.classList.remove('hidden'); } else { throw new Error("Tidak ada gambar yang dihasilkan. Model mungkin tidak dapat membuat gambar berdasarkan prompt Anda, atau batas penggunaan gratis telah tercapai."); } } catch (error) { console.error('Error generating image:', error); // Menampilkan pesan yang lebih user-friendly untuk masalah billing/kuota if (error.message.includes('status: 400') && error.message.includes('billed users')) { showMessage("Gagal membuat gambar. API yang lebih canggih (Imagen/Video API) mungkin memerlukan akun yang mengaktifkan penagihan. Harap periksa kuota penggunaan gratis Anda."); } else { showMessage("Gagal membuat gambar: " + error.message); } } finally { loadingIndicator.classList.add('hidden'); generateImageBtn.disabled = false; } } // --- AKHIR BAGIAN generateImage() YANG DIPERBARUI ---
